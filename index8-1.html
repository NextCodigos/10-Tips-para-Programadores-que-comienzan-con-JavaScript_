Currying

Funciones curriadas


    // Función no curriada
    function suma(x, y) {
      return x + y;
    }
    
    // Función curriada
    function sumaCurriada(x) {
      return function(y) {
        return x + y;
      }
    }
    
    // Función curriada de restar
    function restaCurriada(x) {
      return function(y) {
        return x - y;
      }
    }
    
    // Función curriada para multiplicar por 2
    function multiplicarPorDosCurriada(x) {
      return function() {
        return x * 2;
      }
    }
    
    // Uso
    const sumaTres = sumaCurriada(3);
    console.log(sumaTres(5)); // Salida: 8

    const restaCinco = restaCurriada(5);
    console.log(restaCinco(2)); // Salida: 3

    const multiplicarSeisPorDos = multiplicarPorDosCurriada(6);
    console.log(multiplicarSeisPorDos()); // Salida: 12
    
    // Mostrar resultado en pantalla
    document.getElementById("resultado").innerHTML += "La suma es: " + sumaTres(5) + "
";
    document.getElementById("resultado").innerHTML += "La resta es: " + restaCinco(2) + "
";
    document.getElementById("resultado").innerHTML += "Multiplicar por 2 da: " + multiplicarSeisPorDos() + "
";
