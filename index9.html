Composition

Función de composición


    // Función que duplica un número
    function duplicar(x) {
      return x * 2;
    }
    
    // Función que suma 3 a un número
    function sumarTres(x) {
      return x + 3;
    }
    
    // Función de composición
    function composicion(x) {
      return sumarTres(duplicar(x));
    }
    
    // Uso
    const resultado = composicion(5);
    console.log(resultado); // Salida: 13
    
    // Mostrar resultado en pantalla
    document.getElementById("resultado").innerHTML += "El resultado es: " + resultado;
