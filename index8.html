Currying

Función curriada


    // Función no curriada
    function suma(x, y) {
      return x + y;
    }
    
    // Función curriada
    function sumaCurriada(x) {
      return function(y) {
        return x + y;
      }
    }
    
    // Uso
    const sumaTres = sumaCurriada(3);
    console.log(sumaTres(5)); // Salida: 8
    
    // Mostrar resultado en pantalla
    document.getElementById("resultado").innerHTML += "La suma es: " + sumaTres(5);
