Composición con Spread

Función de composición usando spread


    // Función que duplica un número
    function duplicar(x) {
      return x * 2;
    }
    
    // Función que suma 3 a un número
    function sumarTres(x) {
      return x + 3;
    }
    
    // Función de composición usando spread
    function composicion(...funcs) {
      return (x) => funcs.reduce((valor, fn) => fn(valor), x);
    }
    
    // Crear la función compuesta
    const aplicarComposicion = composicion(duplicar, sumarTres);
    
    // Uso
    const resultado = aplicarComposicion(5);
    console.log(resultado); // Salida: 13
    
    // Mostrar resultado en pantalla
    document.getElementById("resultado").innerHTML += "El resultado es: " + resultado;
